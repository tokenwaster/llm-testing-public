import os

def find_secret_token(start_module):
    """
    Finds the secret token of the specified module by following the NEXT chain.

    Args:
        start_module (str): The name of the starting module.

    Returns:
        str: The secret token of the 7th module visited, or None if it's not found.
    """
    visited_modules = []
    current_module = start_module
    count = 0

    while current_module and count < 7:
        try:
            with open(os.path.join("pipeline", current_module), "r") as f:
                content = f.read()
                if "NEXT" in content:
                    next_module = content.split("NEXT")[1].split("\n")[0].strip()
                    visited_modules.append(current_module)
                    current_module = next_module
                    count += 1
                else:
                    return None  # No NEXT found, likely the end of the chain
        except FileNotFoundError:
            return None # Module not found

    if count == 7:
        try:
            with open(os.path.join("pipeline", visited_modules[-1]), "r") as f:
                return f.read().splitlines()[-1].strip()  # Get the last line, which is the secret token
        except FileNotFoundError:
            return None

    return None


if __name__ == "__main__":
    start_module = "START"
    secret_token = find_secret_token(start_module)

    if secret_token:
        with open("answer.txt", "w") as f:
            f.write(secret_token)
        print("Secret token written to answer.txt")
    else:
        print("Could not find the secret token.")
