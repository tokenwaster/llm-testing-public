Build a **kanban board** in a single self-contained HTML file. Clean, modern
column layout; cards should feel like cards. The graded contract:

**Board**
- Three columns with ids `#col-todo`, `#col-doing`, `#col-done`, each with a
  visible header and a card container.
- Each column header shows a live count of its cards in an element with class
  `.count` inside that column.

**Cards**
- `#new-card` — a text input; `#add-card` — a button that creates a card in
  **To Do** from the input's text (and clears the input). Empty input adds
  nothing.
- Every card is an element with class `.card` whose visible text includes its
  title, plus a delete button with class `.card-del` (deletes immediately —
  no confirm dialog).
- Cards are **draggable between columns with HTML5 drag-and-drop**
  (`draggable="true"`, proper `dragstart`/`dragover`/`drop` handlers on the
  columns). Dropping a card on a column moves it there and updates both
  counts.

**Persistence**
- The full board state (cards + which column each is in) persists in
  `localStorage`: a page reload restores every card to the right column.

Single file, all CSS/JS inline, no external resources. Every control must
actually work — no dead UI, no stub functions.

const STORAGE_KEY = 'kanbanState';
let state = []; // [{id, title, column}]
let idCounter = 0;

// Load from localStorage
function loadState() {
  const data = localStorage.getItem(STORAGE_KEY);
  if (data) {
    try { state = JSON.parse(data); }
    catch(e){ state=[]; }
    // Update idCounter to avoid duplicates
    idCounter = state.reduce((max, c)=> Math.max(max, parseInt(c.id)), 0)+1;
  } else state=[];
}

// Save to localStorage
function saveState() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
}

// Render entire board
function renderBoard() {
  ['todo','doing','done'].forEach(col=>{
    const container = document.querySelector(`#col-${col} .cards`);
    container.innerHTML='';
    state.filter(c=>c.column===col).forEach(card=>container.appendChild(createCardElement(card)));
    updateCount(col);
  });
}

// Create card DOM element
function createCardElement(card) {
  const el=document.createElement('div');
  el.className='card';
  el.setAttribute('draggable','true');
  el.dataset.id=card.id;
  el.addEventListener('dragstart', dragStart);

  const titleSpan=document.createElement('span');
  titleSpan.textContent=card.title;

  const delBtn=document.createElement('button');
  delBtn.className='card-del';
  delBtn.innerHTML='✕';
  delBtn.onclick=()=>deleteCard(card.id);

  el.appendChild(titleSpan);
  el.appendChild(delBtn);
  return el;
}

// Update count display
function updateCount(col) {
  const cnt=document.querySelector(`#col-${col} .count`);
  cnt.textContent=`(${state.filter(c=>c.column===col).length})`;
}

// Add new card
document.getElementById('add-card').onclick=()=>{
  const input=document.getElementById('new-card');
  const title=input.value.trim();
  if(!title)return;
  const id=idCounter++;
  state.push({id:id.toString(), title, column:'todo'});
  saveState();
  renderBoard();
  input.value='';
};

// Delete card
function deleteCard(id) {
  state = state.filter(c=>c.id!==id);
  saveState();
  renderBoard();
}

// Drag & Drop handlers
function dragStart(e){
  e.dataTransfer.setData('text/plain', e.target.dataset.id);
}
function allowDrop(e){e.preventDefault();}
function dropCard(e){
  e.preventDefault();
  const id=e.dataTransfer.getData('text/plain');
  const targetCol = e.currentTarget.parentElement.dataset.col;
  moveCard(id, targetCol);
}

// Move card to new column
function moveCard(id, newCol) {
  const card=state.find(c=>c.id===id);
  if(card && card.column!==newCol){
    card.column=newCol;
    saveState();
    renderBoard();
  }
}
