            txn_id = int(match.group(1))
            voided_txns.add(txn_id)
        continue
    
    # Check for AMEND

        match = re.match(r'\[AMEND (\d+) ([+-]?\d+)\]', line)
        if match:
            txn_id = int(match.group(1))
            amount = int(match.group(2))
            amendments[txn_id] = amount
        continue
    
    # Check for TXN
    if line.startswith('[TXN'):

            txn_id = int(match.group(1))
            account = match.group(2)
            amount = int(match.group(3))
            status = match.group(4)

            }
